public class BinarySearchTree {
private Node insertNodeRecursive(Node head, int data) {
if (head == null) {
return new Node(data);
} else if (data > head.data) {
head.right = insertNodeRecursive(head.right, data);
} else if (data <= head.data) {
head.left = insertNodeRecursive(head.left, data);
}
return head;
}
}
